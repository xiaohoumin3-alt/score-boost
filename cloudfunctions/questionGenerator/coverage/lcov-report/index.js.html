/**
 * 获取待处理的队列任务
 * @param {Object} db - 数据库实例
 * @param {number} maxTasks - 最大处理任务数
 * @returns {Promise<Array>} 待处理任务列表
 */
async function fetchPendingTasks(db, maxTasks = 3) {
  try {
    const result = await db.collection('question_queue')
      .where({ status: 'pending' })
      .orderBy('priority', 'desc')
      .orderBy('created_at', 'asc')
      .limit(maxTasks)
      .get();
 
    return result.data || [];
  } catch (e) {
    console.error('[fetchPendingTasks] Error:', e);
    return [];
  }
}
 
/**
 * 更新队列任务状态
 * @param {Object} db - 数据库实例
 * @param {string} queueId - 队列任务ID
 * @param {string} status - 新状态
 * @param {Object} extraFields - 额外字段
 */
async function updateQueueStatus(db, queueId, status, extraFields = {}) {
  try {
    const updateData = {
      status,
      updated_at: new Date().toISOString(),
      ...extraFields
    };
 
    await db.collection('question_queue').doc(queueId).update({
      data: updateData
    });
 
    return { success: true };
  } catch (e) {
    console.error('[updateQueueStatus] Error:', e);
    return { success: false, error: e.message };
  }
}
 
/**
 * 检查任务是否被取消
 * @param {Object} db - 数据库实例
 * @param {string} queueId - 队列任务ID
 * @returns {Promise<boolean>} 是否被取消
 */
async function checkTaskCancelled(db, queueId) {
  try {
    const result = await db.collection('question_queue').doc(queueId).get();
    const task = result.data;
    return !!(task && task.status === 'cancelled');
  } catch (e) {
    return false;
  }
}

 * 为任务生成题目
 * @param {Object} task - 队列任务
 * @param {Function} generateAi - AI生成函数
 * @param {Object} db - 数据库实例（用于中断检测）
 * @returns {Promise<Array>} 生成的题目列表
 */
async function generateQuestionsForTask(task, generateAi, db = null) {
  const { num_questions, difficulty_distribution, subject, grade, semester, _id } = task;
  const allQuestions = [];
 
  // 计算每个难度的题目数量，处理缺失 difficulty_distribution
  // 注意：必须使用 typeof 检查，因为 0 是有效值但 0 || x 会返回 x
  const dist = difficulty_distribution || { easy: 0.5, medium: 0.3, hard: 0.2 };
  const easyCount = Math.round(num_questions * (typeof dist.easy === 'number' ? dist.easy : 0.5));
  const mediumCount = Math.round(num_questions * (typeof dist.medium === 'number' ? dist.medium : 0.3));
  const hardCount = num_questions - easyCount - mediumCount;
 
  const difficulties = [
    { level: 'easy', count: easyCount },
    { level: 'medium', count: mediumCount },
    { level: 'hard', count: hardCount }
  ];
 
  for (const { level, count } of difficulties) {
    if (count <= 0) continue;
 
    // 中断检测：检查任务是否被取消
    if (db && _id) {
      const cancelled = await checkTaskCancelled(db, _id);
      if (cancelled) {
        console.log(`[generateQuestions] Task ${_id} was cancelled, stopping generation`);
        throw new Error('TASK_CANCELLED');
      }
    }
 
    try {
      const questions = await generateAi(task, level, count);
      allQuestions.push(...questions);
 
      // 更新进度前先检查状态，防止覆盖 cancelled 状态
      if (db && _id) {
        const cancelled = await checkTaskCancelled(db, _id);
        if (cancelled) {
          console.log(`[generateQuestions] Task ${_id} was cancelled before progress update`);
          throw new Error('TASK_CANCELLED');
        }
 
        const generatedCount = allQuestions.length;
        await updateQueueStatus(db, _id, 'processing', {
          progress: {
            generated: generatedCount,
            total: num_questions,
            percent: Math.floor((generatedCount / num_questions) * 100)
          }
        });
      }
    } catch (e) {
      if (e.message === 'TASK_CANCELLED') {
        throw e; // 重新抛出取消错误
      }
      console.error(`[generateQuestions] Error generating ${level} questions:`, e);
      throw e;
    }
  }
 
  return allQuestions;
}

/**

 * @param {number} count - 题目数量
 * @returns {Promise<Array>} 生成的题目
 */

    const duration = Date.now() - startTime;
    console.log(`[generateAi] SUCCESS task:${task._id} difficulty:${difficulty} count:${questions.length} duration:${duration}ms`);

    console.error(`[generateAi] FAILED task:${task._id} difficulty:${difficulty} duration:${duration}ms error:`, e.message);
    throw e;
  }
}
