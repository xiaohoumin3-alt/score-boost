function safeFetch(url, options) {

      }
    }
 
    const timeout = options.timeout || 60000;
 
    const reqOptions = {
      hostname: urlObj.hostname,

      path: urlObj.pathname + urlObj.search,
      method: options.method || 'GET',
      headers: headers,
      timeout: timeout
    };
 
    const req = protocol.request(reqOptions, (res) => {
      let body = '';
      res.on('data', (chunk) => body += chunk);
      res.on('end', () => {

          status: res.statusCode,
          text: async () => body,
          json: async () => JSON.parse(body)

      reject(err);

/**
 * LaTeX 到 Unicode 转换（后处理）
 * 清理AI生成时可能残留的LaTeX格式
 */
function postProcessLatex(text) {
  if (!text || typeof text !== 'string') return text;
 
  let result = text;
 
  // 移除 $ 符号
  result = result.replace(/\$/g, '');
 
  // 分数（先处理，避免与花括号冲突）
  result = result.replace(/\\frac\{([^{}]+)\}\{([^{}]+)\}/g, '$1/$2');
 
  // 立方根
  result = result.replace(/\\sqrt\[3\]\{([^{}]+)\}/g, '³√($1)');
 
  // 平方根
  result = result.replace(/\\sqrt\{([^{}]+)\}/g, '√($1)');
 
  // 希腊字母和符号
  result = result.replace(/\\pi/g, 'π');
  result = result.replace(/\\leq/g, '≤');
  result = result.replace(/\\geq/g, '≥');
  result = result.replace(/\\neq/g, '≠');
  result = result.replace(/\\times/g, '×');
  result = result.replace(/\\div/g, '÷');
  result = result.replace(/\\pm/g, '±');
  result = result.replace(/\\infty/g, '∞');
  result = result.replace(/\\angle/g, '∠');
  result = result.replace(/\\circ/g, '°');
  result = result.replace(/\\parallel/g, '∥');
  result = result.replace(/\\perp/g, '⊥');
  result = result.replace(/\\triangle/g, '△');
 
  // 清理剩余的花括号
  result = result.replace(/\{([^{}]+)\}/g, '($1)');
 
  return result;
}

/**
 * 洗牌算法（Fisher-Yates）
 * @param {Array} array - 要打乱的数组
 * @returns {Array} 打乱后的数组
 */
function shuffleArray(array) {
  const result = [...array];
  for (let i = result.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [result[i], result[j]] = [result[j], result[i]];
  }
  return result;
}

      ? await getExistingQuestions(kp.kp_id, 10)
      : [];
 
    console.log('[ENTRY] RAG context:', kc.knowledge_context ? 'present' : 'empty');
    console.log('[ENTRY] Existing questions:', existingQuestions.length);
 
    // 确定题型
    const questionType = eventData.question_type || 'choice';
    // 确定是否跳过图片生成（用于快速批量生成场景）
    const skipImage = skip_image || false;
 
    console.log('[ENTRY] Calling generateQuestion..., skipImage:', skipImage);
    const question = await generateQuestion(kp, difficulty, {
      question_type: questionType,
      knowledge_context: kc.knowledge_context,
      exclude_questions: existingQuestions,
      skip_image: skipImage
    });
    console.log('[ENTRY] Question generated:', !!question);
 
    // 写入ai_question_pool
    if (database) {
      try {
        const now = new Date().toISOString();
        const record = {
          ...question,
          kp_id: kp.kp_id,
          kp_name: kp.kp_name,
          chapter: kp.chapter || '',
          difficulty,
          question_type: question.question_type,
          verified: false,
          usage_count: 0,
          correct_count: 0,
          created_at: now,
          updated_at: now
        };
        const result = await database.collection('ai_question_pool').add({ data: record });
        console.log('[ENTRY] Question saved to ai_question_pool, id:', result._id);
        question.pool_id = result._id;
        question.saved = true;
      } catch (e) {
        console.error('[ENTRY] Failed to save to ai_question_pool:', e.message);
        question.saved = false;
      }
    } else {
      console.log('[ENTRY] DB not available, skipping save');
      question.saved = false;
    }
 
    return {
      success: true,
      data: question
    };
 
  } catch (e) {
    console.error('[ERROR] generateAiQuestion error:', e.message);
    console.error('[ERROR] Stack:', e.stack);
    return {
      success: false,
      error: e.message
    };
  }
};
 
/**
 * 批量生成题目
 * @param {Array<Object>} tasks - 生成任务列表
 * @param {Object} options - 选项 { skip_image: boolean }
 * @returns {Promise<Array>} 生成的题目列表
 */
async function generateQuestionBatch(tasks, options = {}) {
  const { skip_image = true } = options;
  const results = [];
 
  for (const task of tasks) {
    try {
      const question = await generateQuestion(task.kp, task.difficulty, {
        question_type: task.question_type || 'choice',
        skip_image: skip_image
      });
      results.push(question);
    } catch (e) {
      console.error('[generateQuestionBatch] Failed to generate question for', task.kp?.kp_name, ':', e.message);
      results.push(null);
    }
  }
 
  return results.filter(Boolean);
}
